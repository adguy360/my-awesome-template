# telegram bot api
https://core.telegram.org/bots/api#getupdates

$TELEGRAM_API_TOKEN

http https://api.telegram.org/bot$TELEGRAM_API_TOKEN/getUpdates | jq
// an update is considered confirmed as soon as getUpdates
// is called with an offset higher than its update_id
http https://api.telegram.org/bot$TELEGRAM_API_TOKEN/getUpdates?offset=60598192

http https://api.telegram.org/bot$TELEGRAM_API_TOKEN/getUpdates | jq '.result |  map({"from_id": .message.from.id, "update_id": .update_id, "message_text": .message.text})'
http https://api.telegram.org/bot$TELEGRAM_API_TOKEN/getUpdates | jq '.result[] | select(.message.from.id==$TELEGRAM_FROM_ID)'

# filter message by user
http https://api.telegram.org/bot$TELEGRAM_API_TOKEN/getUpdates | jq '[ .result[] | select(.message.from.id==$TELEGRAM_FROM_ID) ] | map({"update_id": .update_id, "message_text": .message.text})'
